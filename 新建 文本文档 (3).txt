        this._spanArr = {}; // 数据为空时重置
        return [];
      }
      // 深拷贝数据
      const data = JSON.parse(JSON.stringify(this.postList));
      // 合并配置
      const mergeConfig = {
        'structureParent': 'structureParentId',
        'structureFocus': 'structureFocusId',
        'structureChild': 'structureChildId',
        'functionParent': 'functionParentId',
        'functionFocus': 'functionFocusId',
        'functionChild': 'functionChildId',
        'failureEffect': 'failureEffectId',
        'failureSeverity': 'failureSeverity',
        'failureMode': 'failureModeId',

        'riskRpn': 'riskStatusId',
        'riskAp': 'riskStatusId',
        'optAp': 'optRiskStatusId',
        'optDetection': 'optRiskStatusId',
        'optDetectionDesc': 'optRiskStatusId',
        'optOccurrence': 'optRiskStatusId',
        'optPreventDesc': 'optRiskStatusId',
        'optRpn': 'optRiskStatusId'
      };
      const strictMergeFields = ['riskStatusId', 'optRiskStatusId'];
      // 初始化 spanArr
      const spanArr = {};
      Object.keys(mergeConfig).forEach(prop => {
        spanArr[prop] = [];
      });
      // 计算合并规则并合并数组
      Object.keys(mergeConfig).forEach(prop => {
        const idField = mergeConfig[prop];
        const isStrict = strictMergeFields.includes(idField);
        let pos = 0;
        data.forEach((item, index) => {
          if (index === 0) {
            spanArr[prop].push(1);
            pos = 0;
          } else {
            const currentItem = data[index];
            const prevItem = data[index - 1];
            // 获取当前和上一个的ID值
            const currentId = currentItem[idField];
            const prevId = prevItem[idField];
             // 修改判断逻辑：当前ID存在且下条数据ID不存在时也合并
            const isIdSame = (currentId === prevId) || (!currentId && prevId);
            let isFailureReasonSame = true; 
            if (isStrict) {
              const currentFrId = currentItem.failureReasonId;
              const prevFrId = prevItem.failureReasonId;
              isFailureReasonSame = (currentFrId === prevFrId) || (!currentFrId && prevFrId);
            }
            const isKeySame = currentItem.mergeKey === prevItem.mergeKey;
            if (isIdSame && isKeySame && isFailureReasonSame)  {
              spanArr[prop][pos] += 1;
              spanArr[prop].push(0);
            } else {
              spanArr[prop].push(1);
              pos = index;
            }
          }
        });
      });
      // 存储合并规则供 mergeRowMethod 使用
      this._spanArr = spanArr;
      return data;
    }

    showMenu(event) {
      const menu = document.querySelector("#menu");
      event.preventDefault(); //阻止浏览器的默认行为
      // 确保菜单已渲染但隐藏，以便获取尺寸
      menu.style.display = 'block';
      menu.style.visibility = 'hidden';
      // 获取菜单和窗口尺寸
      const menuRect = menu.getBoundingClientRect();
      const menuWidth = menuRect.width;
      const menuHeight = menuRect.height;
      const windowWidth = window.innerWidth;
      const windowHeight = window.innerHeight;
      // 计算理想位置
      let left = event.clientX;
      let top = event.clientY;
      // 调整水平位置防止超出右边界
      if (left + menuWidth > windowWidth) {
        left = windowWidth - menuWidth - 5; // 留5px边距
      }
      // 调整垂直位置防止超出下边界
      if (top + menuHeight > windowHeight) {
        top = windowHeight - menuHeight - 5; // 留5px边距
      }
      // 确保位置不小于0
      left = Math.max(5, left);
      top = Math.max(5, top);
      // 应用最终位置
      menu.style.left = `${left}px`;
      menu.style.top = `${top}px`;
      menu.style.display = 'block';
      menu.style.visibility = 'visible';
      menu.style.zIndex = '9999';

    },

      let treeId
      switch (column.property) {
        case 'structureParent':  //上一级结构
          extraId = row.structureParentId
          break;
        case 'structureFocus': //关注要素
          extraId = row.structureFocusId
          break;
        case 'structureChild': //下一级结构
          extraId = row.structureChildId
          break;
        /* 功能分析 */
        case 'functionParent': //上一级结构的功能特性
          if (row.functionParentList && Array.isArray(row.functionParentList)) {
            treeId = row.functionParentList[0].id
          }
          extraId = row.structureParentId
          break;
        case 'functionFocus': //关注要素的功能特性
          treeId = row.functionFocusId
          extraId = row.structureFocusId
          break;
        case 'functionChild': //下一级结构的功能特性
          treeId = row.functionChildId
          extraId = row.structureChildId
          break;
        /* 失效分析 */
        case 'failureEffect'://失效影响(FE)
          if (row.failureEffectList && Array.isArray(row.failureEffectList)) {
            treeId = row.failureEffectList[0].id
          }
          extraId = row.structureParentId
          break;
        case 'failureMode'://失效模式(FM)
          treeId = row.failureModeId
          extraId = row.structureFocusId
          break;
        case 'failureReason'://失效原因(FC)
          treeId = row.failureReasonId
          extraId = row.structureChildId
          break;
        /* 风险分析 */
        case 'riskPreventDesc'://预防措施(PC)
          treeId = row.riskStatusId
          extraId = row.structureChildId
          break;
        case 'riskOccurrence'://发生度(O)
          treeId = row.riskStatusId
          extraId = row.structureChildId
          break;
        case 'riskDetectionDesc'://探测措施(DC)
          treeId = row.riskStatusId
          extraId = row.structureChildId
          break;
        case 'optPreventDesc'://预防措施(PC)
          treeId = row.optRiskStatusId
          extraId = row.structureChildId
          break;
        case 'optOccurrence'://发生度(O)
          treeId = row.optRiskStatusId
          extraId = row.structureChildId
          break;
        case 'optDetectionDesc'://探测措施(DC)
          treeId = row.optRiskStatusId
          extraId = row.structureChildId
          break;
        case 'optDetection'://探测度(D)
          treeId = row.optRiskStatusId
          extraId = row.structureChildId
          break;

      }
      this.setProjectFocusId(this.$route.params.projectId, extraId)
      this.setRightTreeFocusId(this.$route.params.projectId, treeId)
      this.$emit('tabClick')

          if (res === column.prop) { originIndex = i }
        })
      }
      if (column.order) {
        if (column.order === 'ascending') {
          this.queryParams.params.orderType.push('asc')
        } else if (column.order === 'descending') {
          this.queryParams.params.orderType.push('desc')
        }
        this.queryParams.params.orderName.push(column.prop)
      }
      if (typeof originIndex === 'number') {
        this.queryParams.params.orderName.splice(originIndex, 1)
        this.queryParams.params.orderType.splice(originIndex, 1)
      }
      this.getList()
